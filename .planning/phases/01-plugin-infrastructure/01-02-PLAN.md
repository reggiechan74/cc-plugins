```python
#!/usr/bin/env python3
"""
setup_runtime.py — Install Python runtime dependencies for meta-compiler.

Usage:
    python3 setup_runtime.py [--force]

    --force: Delete and recreate the venv before installing (handles broken venv).

Environment:
    CLAUDE_PLUGIN_ROOT: Plugin root directory (set by Claude Code). If absent,
    falls back to the directory containing this script's parent (meta-compiler/).
"""

import argparse
import shutil
import subprocess
import sys
import os
from pathlib import Path

PACKAGES = ["ortools", "z3-solver", "scipy", "pydantic", "jinja2"]


def get_plugin_root() -> Path:
    """Return the plugin root directory. Uses CLAUDE_PLUGIN_ROOT env var if set,
    otherwise resolves relative to this script's location."""
    env_root = os.environ.get("CLAUDE_PLUGIN_ROOT", "")
    if env_root:
        return Path(env_root)
    # Fallback: this file is at <plugin_root>/scripts/setup_runtime.py
    return Path(__file__).parent.parent


def get_venv_path(plugin_root: Path) -> Path:
    """Return the venv path, falling back to home dir if plugin root is not writable."""
    candidate = plugin_root / ".venv"
    try:
        candidate.parent.mkdir(parents=True, exist_ok=True)
        probe = candidate.parent / ".write_probe"
        probe.touch()
        probe.unlink()
        return candidate
    except (OSError, PermissionError):
        fallback = Path.home() / ".claude" / "meta-compiler" / ".venv"
        print(f"[INFO] Plugin root not writable; using fallback venv: {fallback}")
        return fallback


def create_venv(venv_path: Path) -> None:
    """Create a virtual environment at venv_path."""
    print(f"[INFO] Creating venv at {venv_path}")
    subprocess.run([sys.executable, "-m", "venv", str(venv_path)], check=True)


def check_in_venv(venv_python: Path, package: str) -> str | None:
    """Return installed version string if package is installed in the venv, else None.
    Queries the VENV Python — never the system Python."""
    result = subprocess.run(
        [
            str(venv_python),
            "-c",
            f"import importlib.metadata; print(importlib.metadata.version('{package}'))",
        ],
        capture_output=True,
        text=True,
    )
    return result.stdout.strip() if result.returncode == 0 else None


def install_packages(venv_python: Path, packages: list[str], use_uv: bool) -> None:
    """Install packages into the venv."""
    if use_uv:
        cmd = ["uv", "pip", "install", "--python", str(venv_python)] + packages
    else:
        cmd = [str(venv_python), "-m", "pip", "install"] + packages
    subprocess.run(cmd, check=True)


def run_setup(plugin_root: Path, force: bool = False) -> int:
    """Main setup logic. Returns 0 on success, 1 on failure."""
    use_uv = shutil.which("uv") is not None
    installer_name = "uv" if use_uv else "pip"
    print(f"[INFO] Installer: {installer_name}")

    venv_path = get_venv_path(plugin_root)

    # --force: destroy and recreate
    if force and venv_path.exists():
        print(f"[INFO] --force: removing existing venv at {venv_path}")
        shutil.rmtree(venv_path)

    # Create venv if it doesn't exist
    if not venv_path.exists():
        create_venv(venv_path)
    else:
        print(f"[INFO] Venv exists at {venv_path}")

    venv_python = venv_path / "bin" / "python"

    # Idempotency: check each package before installing
    already_installed = []
    to_install = []
    for pkg in PACKAGES:
        ver = check_in_venv(venv_python, pkg)
        if ver:
            already_installed.append((pkg, ver))
        else:
            to_install.append(pkg)

    for pkg, ver in already_installed:
        print(f"[SKIP] {pkg}=={ver} -- already installed")

    if to_install:
        print(f"[INSTALL] {', '.join(to_install)} (using {installer_name})")
        try:
            install_packages(venv_python, to_install, use_uv)
            print(f"[OK] Installed: {', '.join(to_install)}")
        except subprocess.CalledProcessError as e:
            print(f"[ERROR] Install failed: {e}")
            return 1
    else:
        print("[OK] All dependencies already installed.")

    newly = len(to_install)
    skipped = len(already_installed)
    print(f"\nSummary: {newly} installed, {skipped} already present.")
    return 0


def main() -> None:
    parser = argparse.ArgumentParser(description="Install meta-compiler runtime dependencies.")
    parser.add_argument(
        "--force",
        action="store_true",
        help="Delete and recreate the venv before installing.",
    )
    args = parser.parse_args()

    plugin_root = get_plugin_root()
    sys.exit(run_setup(plugin_root, force=args.force))


if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""
doctor_runtime.py — Diagnose meta-compiler Python runtime environment.

Usage:
    python3 doctor_runtime.py

Output: Terse pass/fail report with version numbers and fix hints for failures.
"""

import os
import subprocess
import sys
from pathlib import Path

PACKAGES = ["ortools", "z3-solver", "scipy", "pydantic", "jinja2"]
# Map PyPI install name to importlib.metadata distribution name (same for all our deps)
DIST_NAMES = {
    "ortools": "ortools",
    "z3-solver": "z3-solver",
    "scipy": "scipy",
    "pydantic": "pydantic",
    "jinja2": "jinja2",
}


def get_plugin_root() -> Path:
    """Return the plugin root directory."""
    env_root = os.environ.get("CLAUDE_PLUGIN_ROOT", "")
    if env_root:
        return Path(env_root)
    return Path(__file__).parent.parent


def find_venv_python(plugin_root: Path) -> Path | None:
    """Return the venv Python path if a venv exists, else None."""
    for candidate in [
        plugin_root / ".venv" / "bin" / "python",
        Path.home() / ".claude" / "meta-compiler" / ".venv" / "bin" / "python",
    ]:
        if candidate.exists():
            return candidate
    return None


def check_python_version() -> tuple[bool, str]:
    """Return (ok, version_string) for the current Python."""
    version = f"{sys.version_info.major}.{sys.version_info.minor}.{sys.version_info.micro}"
    ok = sys.version_info >= (3, 8)
    return ok, version


def check_package_in_venv(venv_python: Path, dist_name: str) -> str | None:
    """Return version string if package is installed in venv, else None."""
    result = subprocess.run(
        [
            str(venv_python),
            "-c",
            f"import importlib.metadata; print(importlib.metadata.version('{dist_name}'))",
        ],
        capture_output=True,
        text=True,
    )
    return result.stdout.strip() if result.returncode == 0 else None


def run_doctor(plugin_root: Path) -> int:
    """Run health checks and print terse pass/fail report. Returns 0 if all pass, 1 if any fail."""
    print("meta-compiler runtime health check")
    print("===================================")

    # Python version
    py_ok, py_version = check_python_version()
    status = "[PASS]" if py_ok else "[FAIL]"
    print(f"{status} python        {py_version}")
    if not py_ok:
        print("       fix: Python 3.8+ required")

    # Find venv
    venv_python = find_venv_python(plugin_root)
    if venv_python is None:
        print("[FAIL] venv          NOT FOUND  ->  run: model:setup")
        print("\nResult: venv missing. Run `model:setup` to fix.")
        return 1

    # Check each package
    failed = []
    for pkg in PACKAGES:
        dist = DIST_NAMES[pkg]
        version = check_package_in_venv(venv_python, dist)
        if version:
            print(f"[PASS] {pkg:<14} {version}")
        else:
            print(f"[FAIL] {pkg:<14} NOT FOUND  ->  run: model:setup")
            failed.append(pkg)

    print()
    if failed:
        count = len(failed)
        print(f"Result: {count} missing. Run `model:setup` to fix.")
        return 1
    else:
        print("Result: all OK.")
        return 0


def main() -> None:
    plugin_root = get_plugin_root()
    sys.exit(run_doctor(plugin_root))


if __name__ == "__main__":
    main()
```
